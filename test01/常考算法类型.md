```
int n;
bool isHe[1600000]; // false表示素数,true表示非素数
int prim[1000005]; // 记录素数的值
int cnt; // 记录素数的个数
int main()
{
cin >> n;
isHe[0] = isHe[1] = true;
//遍历i 到 n,prim数组存放的是到i为止的所有质数
for (int i = 2; i <= n; ++i) {
    if (!isHe[i]) {
        prim[cnt++] = i;
    }
    for (int j = 0; j < cnt && i * prim[j] <= n; ++j) {
        isHe[i * prim[j]] = true;
        if (i % prim[j] == 0) {
            break;
        }
    }
}
cout << cnt << endl;
return 0;
}
```

## 5.前缀和
### (1)k倍区间
```
//k倍区间 ,前缀和的题目
//前缀和的优势是快速地对一段区间求和
int n,k;
long long cnt;
int sum[100005];
int crr[100005];   //这个数组分别记录1 ~ n的前缀和%k == i的数量,比如a[i] == x ,则说明有x种情况余数是i
				   // 相同余数的的区间相减就是一个余数为0的区间
				   // 举个例子，当有10个余数为3的区间的时候,他们构成的余数为0的区间(区间相减)是10 * 9 / 2,也就是 1 + 2 + 3 .... + 9
				   // 所以也可以使用前缀和的方法来计算这一数量
				   // 例外就是当前区间不需要相减就已经余数为0的时候，当有10个余数为0的区间的时候 ,总共可以构成 1 + 2 + 3 ... + 10
				   // 个前缀为0的区间，所以crr[0] 要初始化为1
int main(){
	cin >> n >> k;
	crr[0] = 1;
	for(int i = 1;i<= n;++ i){
		cin >> sum[i];
		sum[i] += sum[i - 1];
		cnt = cnt + crr[sum[i] % k];
		crr[sum[i] % k]++;
	}
	cout << cnt <<endl;

	return 0;
}
```

```
//单调队列，以O(n)的复杂度求每个区间的最值
deque<int>deq;
int n,m;   //总共多少个数字，区间长度
int arr[100005];
//先求最小值
int main(){
	cin >> n >> m;
		for(int i = 1;i <= n;++i){
		cin >> arr[i];
	}
	for(int i = 1;i <= n;++i){
		//删除过期的数字
		while(!deq.empty() && deq.front() < i - m + 1){
			deq.pop_front();
		}
		//如果新来的数字更最小，就删除原来的所有数字
		while(!deq.empty() && arr[deq.front()] >= arr[i]){
			deq.pop_front();
		}
		//加入数字到队列并且打印当前区间的最小值
		deq.push_back(i);
		if(i >= m){
			cout << arr[deq.front()] <<endl;
		}
	}
	return 0;
}
```

## 12.边界二分

### (1)二分边界,右边界
```
int arr[15] = {1,4,4,4,4,6,16,17,17,17,17,20,30,30,30};

int main(){
	int tar;
	cin >> tar;
	int l = 0,r = 14, m;
	while(l < r){
		m = (l + r + 1) / 2;   //右边界需要+1,否则一些情况下会进入无限循环
		if(arr[m] > tar){
			r = m - 1;
		}else{
			l = m;    //等于也需要向右走
		}
	}
	cout << "找到的tar在数组中的下标是:" << r <<endl;
	
	return 0;
}
```

## 13.广度优先搜索bfs
```
ACWing 献给阿尔吉侬的花束 BFS学习
int T;
int R, C;
struct Point {
    int x;
    int y;
};
int dx[4] = { 0, 0, 1, -1 };
int dy[4] = { -1, 1, 0, 0 };

char land[205][205];
int maplen[205][205]; // 记录步长的数组
std::queue<Point> que;
int main()
{
    cin >> T;
    while (T--) {
        memset(maplen, 127, sizeof(maplen));
        cin >> R >> C;
        for (int i = 0; i < R; ++i) {
            for (int j = 0; j < C; ++j) {
                cin >> land[i][j];
                if (land[i][j] == 'S') {
                    Point p;
                    p.x = i;
                    p.y = j;
                    que.push(p);
                    maplen[i][j] = 0;
                }
            }
        }
        // 遍历找到起点
        bool f = false;
        while (!que.empty()) {
            Point nowp = que.front();
            // cout << "当前坐标:" << nowp.x << "  " << nowp.y << endl;
            que.pop();
            for (int i = 0; i < 4; ++i) {
                int xx = nowp.x + dx[i];
                int yy = nowp.y + dy[i];
                // 如果在地图内并且可以走
                if (xx >= 0 && yy >= 0 && xx < R && yy < C && maplen[xx][yy] > maplen[nowp.x][nowp.y] + 1 && land[xx][yy] != '#') {
                    Point nextp;
                    nextp.x = xx;
                    nextp.y = yy;
                    que.push(nextp);
                    maplen[xx][yy] = maplen[nowp.x][nowp.y] + 1;
                    // cout << xx << "  " << yy << "  " << maplen[xx][yy] << endl;
                    if (land[xx][yy] == 'E') {
                        // cout << "找到终点" << endl;
                        f = true;
                        break;
                    }
                }
            }
            if (f) {
                break;
            }
        }
        for (int i = 0; i < R; ++i) {
            for (int j = 0; j < C; ++j) {
                if (land[i][j] == 'E') {
                    if (maplen[i][j] != 2139062143) {
                        cout << maplen[i][j] << endl;
                    } else {
                        cout << "oop!" << endl;
                    }
                }
            }
        }
    }

    return 0;
}
```
